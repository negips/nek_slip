17 /*
18  When the preprocessor macro MPI is defined, defines (very) thin wrappers
19  for the handful of used MPI routines. Alternatively, when MPI is not defined,
20  these wrappers become dummy routines suitable for a single process run.
21  No code outside of "comm.h" and "comm.c" makes use of MPI at all.
22 
23  Basic usage:
24 
25  struct comm c;
26 
27  comm_init(&c, MPI_COMM_WORLD); // initializes c using MPI_Comm_dup
28 
29  comm_free(&c);
30 
31  Very thin MPI wrappers: (see below for implementation)
32 
33  comm_send,_recv,_isend,_irecv,_time,_barrier
34 
35  Additionally, some reduction and scan routines are provided making use
36  of the definitions in "gs_defs.h" (provided this has been included first).
37 
38  Example comm_allreduce usage:
39 
40  double v[5], buf[5];
41  comm_allreduce(&c, gs_double,gs_add, v,5,buf);
42  // Computes the vector sum of v across all procs, using
43  // buf as a scratch area. Delegates to MPI_Allreduce if possible.
44 
45  Example comm_scan usage:
46 
47  long in[5], out[2][5], buf[2][5];
48  comm_scan(out, &c,gs_long,gs_add, in,5,buf);
49  // out[0] will be the vector sum of "in" across procs with ids
50  *strictly* less than this one (exclusive behavior),
51  and out[1] will be the vector sum across all procs, as would
52  be computed with comm_allreduce.
53  Note: differs from MPI_Scan which has inclusive behavior
54 
55  Example comm_reduce_double, etc. usage:
56 
57  T out, in[10];
58  out = comm_reduce_T(&c, gs_max, in, 10);
59  // out will equal the largest element of "in",
60  across all processors
61  // T can be "double", "float", "int", "long", "slong", "sint", etc.
62  as defined in "gs_defs.h"
63 
64 */
