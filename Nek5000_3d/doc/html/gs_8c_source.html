647  void *data, gs_mode mode, unsigned vn, gs_dom dom, gs_op op,
648  unsigned transpose, const void *execdata, const struct comm *comm, char *buf)
649 {

651  static gs_scatter_fun *const scatter_user_to_buf[] =
652  { &gs_scatter, &gs_scatter_vec, &gs_scatter_many_to_vec, &scatter_noop };
653  static gs_scatter_fun *const scatter_buf_to_buf[] =
654  { &gs_scatter, &gs_scatter_vec, &gs_scatter_vec, &gs_scatter };
655  static gs_scatter_fun *const scatter_buf_to_user[] =
656  { &gs_scatter, &gs_scatter_vec, &gs_scatter_vec_to_many, &scatter_noop };
657  static gs_gather_fun *const gather_buf_to_user[] =
658  { &gs_gather, &gs_gather_vec, &gs_gather_vec_to_many, &gather_noop };
659  static gs_gather_fun *const gather_buf_to_buf[] =
660  { &gs_gather, &gs_gather_vec, &gs_gather_vec, &gs_gather };
661  const unsigned unit_size = vn*gs_dom_size[dom], nstages=crd->nstages;
662  unsigned k;
663  char *sendbuf, *buf_old, *buf_new;
664  const struct cr_stage *stage = crd->stage[transpose];
665  buf_old = buf;
666  buf_new = buf_old + unit_size*crd->stage_buffer_size;
667  /* crystal router */
668  for(k=0;k<nstages;++k) {
669  comm_req req[3];
670  if(stage[k].nrecvn)
671  comm_irecv(&req[1],comm,buf_new,unit_size*stage[k].size_r1,
672  stage[k].p1, comm->np+k);
673  if(stage[k].nrecvn==2)
674  comm_irecv(&req[2],comm,buf_new+unit_size*stage[k].size_r1,
675  unit_size*stage[k].size_r2, stage[k].p2, comm->np+k);
676  sendbuf = buf_new+unit_size*stage[k].size_r;
677  if(k==0)
678  scatter_user_to_buf[mode](sendbuf,data,vn,stage[0].scatter_map,dom);
679  else
680  scatter_buf_to_buf[mode](sendbuf,buf_old,vn,stage[k].scatter_map,dom),
681  gather_buf_to_buf [mode](sendbuf,buf_old,vn,stage[k].gather_map ,dom,op);
682 
683  comm_isend(&req[0],comm,sendbuf,unit_size*stage[k].size_s,
684  stage[k].p1, comm->np+k);
685  comm_wait(&req[0],1+stage[k].nrecvn);
686  { char *t = buf_old; buf_old=buf_new; buf_new=t; }
687  }
688  scatter_buf_to_user[mode](data,buf_old,vn,stage[k].scatter_map,dom);
689  gather_buf_to_user [mode](data,buf_old,vn,stage[k].gather_map ,dom,op);
690 }

695 static uint cr_schedule(struct cr_data *data, const struct comm *comm)
696 {
697  uint mem_size = 0;

699  uint bl=0, n=comm->np;

701  while(n>1) {
702  uint nl = (n+1)/2, bh = bl+nl;
703  if(id<bh) n=nl; else n-=nl,bl=bh;
704  ++k;
705  }
706  data->nstages = k;
707  data->stage[0] = tmalloc(struct cr_stage,2*(k+1));
708  data->stage[1] = data->stage[0] + (k+1);

719  data->stage[1][k].p2 =data->stage[0][k].p2 =comm->id-1;

721  ++k;
722  }
723  return mem_size;
724 }

730 /* assumes sh is grouped by i (e.g., sorted by i or by id) */
731 static void crl_work_init(struct array *cw, struct array *sh,
732  const unsigned send_mask, uint this_p)
733 {
734  const unsigned recv_mask = send_mask^(FLAGS_REMOTE|FLAGS_LOCAL);
735  uint last_i=UINT_MAX; int added_myself;
736  uint cw_n = 0, cw_max = cw->max;
737  struct crl_id *w = cw->ptr;
738  struct shared_id *s, *se;
739 
740 #define CW_ADD(aid,ap,ari,asi) do { \
741  if(cw_n==cw_max) \
742  array_reserve(struct crl_id,cw,cw_n+1),cw_max=cw->max, \
743  w=(struct crl_id*)cw->ptr+cw_n; \
744  w->id=aid, w->p=ap, w->ri=ari, w->si=asi; \
745  ++w, ++cw_n; \
746  } while(0)
747 
748  for(s=sh->ptr,se=s+sh->n;s!=se;++s) {
749  int send = (s->flags&send_mask)==0;
750  int recv = (s->flags&recv_mask)==0;
751  if(s->i!=last_i) last_i=s->i, added_myself=0;
752  if(!added_myself && recv && (s->flags&FLAGS_LOCAL)==0) {
753  added_myself=1;
754  CW_ADD(s->id,this_p,s->i,s->i);
755  }
756  if(send) CW_ADD(s->id,s->p,s->ri,s->i);
757  }
758  cw->n=cw_n;
759 #undef CW_ADD
760 }

762 static uint crl_maps(struct cr_stage *stage, struct array *cw, buffer *buf)
763 {
764  uint mem_size=0;
765  struct crl_id *w, *we, *other;
766  uint scount=1, gcount=1, *sp, *gp;
767  sarray_sort_2(struct crl_id,cw->ptr,cw->n, bi,0, si,0, buf);
768  for(w=cw->ptr,we=w+cw->n;w!=we;w=other) {
769  uint bi=w->bi,any=0,si=w->si;
770  scount+=3;
771  for(other=w+1;other!=we&&other->bi==bi;++other)
772  if(other->si!=si) si=other->si, any=2, ++gcount;
773  gcount+=any;
774  }
775  stage->scatter_map = sp = tmalloc(uint,scount+gcount);
776  stage->gather_map = gp = sp + scount;
777  mem_size += (scount+gcount)*sizeof(uint);
778  for(w=cw->ptr,we=w+cw->n;w!=we;w=other) {
779  uint bi=w->bi,any=0,si=w->si;
780  *sp++ = w->si, *sp++ = bi;
781  *gp++ = bi;
782  for(other=w+1;other!=we&&other->bi==bi;++other)
783  if(other->si!=si) si=other->si, any=1, *gp++ = si;
784  if(any) *gp++ = UINT_MAX; else --gp;
785  *sp++ = UINT_MAX;
786  }
787  *sp=UINT_MAX, *gp=UINT_MAX;
788  return mem_size;
789 }

791 static uint crl_work_label(struct array *cw, struct cr_stage *stage,
792  uint cutoff, int send_hi, buffer *buf,
793  uint *mem_size)
794 {
795  struct crl_id *w, *we, *start;
796  uint nsend, nkeep = 0, nks = 0, bi=0;
797  /* here w->send has a reverse meaning */
798  if(send_hi) for(w=cw->ptr,we=w+cw->n;w!=we;++w) w->send = w->p< cutoff;
799  else for(w=cw->ptr,we=w+cw->n;w!=we;++w) w->send = w->p>=cutoff;
800  sarray_sort_2(struct crl_id,cw->ptr,cw->n, id,1, send,0, buf);
801  for(start=cw->ptr,w=start,we=w+cw->n;w!=we;++w) {
802  nkeep += w->send;
803  if(w->id!=start->id) start=w;
804  if(w->send!=start->send) w->send=0,w->bi=1, ++nks; else w->bi=0;
805  }
806  nsend = cw->n-nkeep;
807  /* assign indices; sent ids have priority (hence w->send is reversed) */
808  sarray_sort(struct crl_id,cw->ptr,cw->n, send,0, buf);
809  for(start=cw->ptr,w=start,we=w+nsend+nks;w!=we;++w) {
810  if(w->id!=start->id) start=w, ++bi;
811  if(w->bi!=1) w->send=1; /* switch back to the usual semantics */
812  w->bi = bi;
813  }
814  stage->size_s = nsend+nks==0 ? 0 : bi+1;
815  for(we=(struct crl_id*)cw->ptr+cw->n;w!=we;++w) {
816  if(w->id!=start->id) start=w, ++bi;
817  w->send = 0; /* switch back to the usual semantics */
818  w->bi = bi;
819  }
820  stage->size_sk = cw->n==0 ? 0 : bi+1;
821  *mem_size += crl_maps(stage,cw,buf);
822  return nsend;
823 }

833 static uint cr_learn(struct array *cw, struct cr_stage *stage,
834  const struct comm *comm, buffer *buf, uint *mem_size)
835 {
836  comm_req req[3];
837  const uint id = comm->id;
838  uint bl=0, n=comm->np;
839  uint size_max=0;
840  uint tag = comm->np;
841  while(n>1) {
842  uint nl = (n+1)/2, bh = bl+nl;
843  uint nkeep, nsend[2], nrecv[2][2] = {{0,0},{0,0}};
844  struct crl_id *wrecv[2], *wsend;
845  nsend[0] = crl_work_label(cw,stage,bh,id<bh,buf, mem_size);
846  nsend[1] = stage->size_s;
847  nkeep = cw->n - nsend[0];
848 
849  if(stage->nrecvn ) comm_irecv(&req[1],comm,nrecv[0],2*sizeof(uint),
850  stage->p1,tag);
851  if(stage->nrecvn==2) comm_irecv(&req[2],comm,nrecv[1],2*sizeof(uint),
852  stage->p2,tag);
853  comm_isend(&req[0],comm,nsend,2*sizeof(uint),stage->p1,tag);
854  comm_wait(req,1+stage->nrecvn),++tag;
855 
856  stage->size_r1 = nrecv[0][1], stage->size_r2 = nrecv[1][1];
857  stage->size_r = stage->size_r1 + stage->size_r2;
858  stage->size_total = stage->size_r + stage->size_sk;
859  if(stage->size_total>size_max) size_max=stage->size_total;
860 
861  array_reserve(struct crl_id,cw,cw->n+nrecv[0][0]+nrecv[1][0]);
862  wrecv[0] = cw->ptr, wrecv[0] += cw->n, wrecv[1] = wrecv[0]+nrecv[0][0];
863  wsend = cw->ptr, wsend += nkeep;
864  if(stage->nrecvn )

869  stage->p2,tag);

876  if(stage->nrecvn==2) crl_bi_to_si(wrecv[1],nrecv[1][0],stage->size_r1);

882  ++stage;
883  }

887 }
